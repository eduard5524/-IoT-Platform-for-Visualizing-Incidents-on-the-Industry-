import {Component, OnInit} from '@angular/core';
import {Router, ActivatedRoute} from '@angular/router';
import {FormBuilder, FormGroup, Validators} from '@angular/forms';
import {AuthenticationService} from '../../services/authentication.service';
import {Md5} from 'md5-typescript';

/**
 * This Component has a login Form with Validation. Validates if user and password are empty before send the request.
 * If valid user create the token and go to USERS LIST by default.
 */

@Component({templateUrl: 'login.component.html'})
export class LoginComponent implements OnInit {
  loginForm: FormGroup;
  loading = false;
  submitted = false;
  returnUrl: string;
  error = '';

  /**
   * Injects formBuilder for validate Forms.
   * Injects route and router for navigation and protect routes by token.
   * Injects authenticationService for login or logout calls.
   */
  constructor(
    private formBuilder: FormBuilder,
    private route: ActivatedRoute,
    private router: Router,
    private authenticationService: AuthenticationService) {
  }

  /**
   * Set the fields on form to validate. Validate if user and password are empty.
   * Reset the token every call by logout() method.
   * Get return url from route parameters
   */
  ngOnInit() {
    this.loginForm = this.formBuilder.group({
      username: ['', Validators.required],
      password: ['', Validators.required]
    });

    // reset login status
    this.authenticationService.logout();

    // get return url from route parameters or default to '/'
    this.returnUrl = this.route.snapshot.queryParams['returnUrl'] || '/users';
  }

  /**
   * Convenience getter for easy access to form fields
   */
  get f() {
    return this.loginForm.controls;
  }

  /**
   * If form submit execute the validation. Shows the loading spinner.
   * If valid form, call to authenticationService passing email and password to API.
   * If valid user create the token and go to USERS LIST by default.
   * Show error messages if API fails or user not valid.
   */
  onSubmit() {
    this.submitted = true;

    // stop here if form is invalid
    if (this.loginForm.invalid) {
      return;
    }

    this.loading = true;
    this.authenticationService.login(this.f.username.value, Md5.init(this.f.password.value))
      .subscribe(
        data => {
          if (data.token) {
            // go to main URL if token exists
            this.authenticationService.setToken(data.token);
            this.router.navigate([this.returnUrl]);
          } else {
            // show the error message if credentials are invalid
            this.error = data.msg;
            this.loading = false;
          }
        },
        error => {
          // show error in console if API sends error
          this.error = error;
          this.loading = false;
          console.log(error);

        });
  }
}
